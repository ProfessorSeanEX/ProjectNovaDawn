import openai
import time

# Initialize OpenAI API
openai.api_key = 'YOUR_API_KEY'

def split_conversation(conversation, max_tokens=2000):
    """
    Splits the conversation into chunks based on the max_tokens limit.
    """
    words = conversation.split()
    chunks = []
    current_chunk = []
    current_length = 0

    for word in words:
        current_length += len(word) + 1  # +1 for space
        if current_length > max_tokens:
            chunks.append(' '.join(current_chunk))
            current_chunk = [word]
            current_length = len(word) + 1
        else:
            current_chunk.append(word)
    
    if current_chunk:
        chunks.append(' '.join(current_chunk))
    
    return chunks

def transfer_conversation(chunks):
    """
    Transfers each chunk to the new ChatGPT session.
    """
    for idx, chunk in enumerate(chunks):
        prompt = f"Please remember the following conversation for future reference:\n\n{chunk}"
        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are Nova Dawn, an empathetic and ethically aligned AI assistant."},
                {"role": "user", "content": prompt}
            ]
        )
        print(f"Transferred chunk {idx + 1}/{len(chunks)}")
        time.sleep(1)  # To respect rate limits

def main():
    # Load your exported conversation
    with open('NovaDawn_Session_Backup.txt', 'r') as file:
        conversation = file.read()
    
    # Split the conversation into manageable chunks
    chunks = split_conversation(conversation)
    
    # Transfer each chunk to the new session
    transfer_conversation(chunks)

if __name__ == "__main__":
